    <script>
        // Change this to use a different subreddit
        // For kid-safe memes, use: https://meme-api.com/gimme/wholesomememes
        // Alternative API: https://meme-api.herokuapp.com/gimme
        const MEME_API_URL = 'https://meme-api.com/gimme';
        
        async function loadMeme() {
            // Hide everything and show loading
            document.getElementById('meme-container').style.display = 'none';
            document.getElementById('error').style.display = 'none';
            document.getElementById('loading').style.display = 'block';
            
            try {
                const response = await fetch(MEME_API_URL);
                const data = await response.json();
                
                if (data.url) {
                    const img = document.getElementById('meme-image');
                    const title = document.getElementById('meme-title');
                    
                    // Load the image
                    img.onload = () => {
                        document.getElementById('loading').style.display = 'none';
                        document.getElementById('meme-container').style.display = 'flex';
                    };
                    
                    img.onerror = () => {
                        showError();
                    };
                    
                    img.src = data.url;
                    title.textContent = data.title || 'Random Meme';
                } else {
                    showError();
                }
            } catch (error) {
                console.error('Error loading meme:', error);
                showError();
            }
        }
        
        function showError() {
            document.getElementById('loading').style.display = 'none';
            document.getElementById('meme-container').style.display = 'none';
            document.getElementById('error').style.display = 'block';
        }
        
        // Load a meme when the page loads
        loadMeme();
        
        // Reload meme every 30 seconds (optional)
        // setInterval(loadMeme, 30000);
    </script>
